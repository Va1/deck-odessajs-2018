              </li>
            <% } %>

        </script>

(2013-2015)

Example: React 

        render () {
          const { fetchItems } = this
          const { items, error } = this.state
         
          return (
            <div>
              <button onClick={fetchItems}>
                Fetch Items
              </button>
              { error === null
                ? <ItemsList items={items} />
                : <Error error={error} /> }
            </div>
          )
        }
      
        state = { items: [], error: null }
         
        fetchItems () {
          $.get('/items')
            .done(items => {
              this.setState({ items, error: null })
            })
            .error(error => {
              this.setState({ items: [], error })
            })
        }

Purge the Unclean!

from impure to pure dataflow

Reactive, Stateful, Pure

(2015-)

Get to know "side-effects"

  • should be separated to keep functions & dataflow pure
  • pure functions & dataflow are much easier to test
  • main examples in web frontend are: backend api requests (AJAX), third-party api requests through iframe (e.g., OAuth SDKs)
  • best known in web frontend as part of Redux (FLUX) -like dataflow/architecture
  • main implementations in web frontend are: Redux-Thunk, Redux-Saga, Redux-Observable, NgRx

Get to know "side-effects"

Get to know Rx

  • focuses on onservable streams concept
  • includes more than 60 operators for stream manipulation
  • made for concurrency and asynchronicity, with reactive & functional programming principles in mind
  • available for JavaScript, Java, Scala, C#, C++, Clojure, Python and many others
  • is an important dependency of Angular 2 toolset

        function mapStateToProps (state) {
          const { items, error } = this.state
          return { items, error }
        }
        const mapDispatchToProps = { fetchItems }
         
        connect(mapStateToProps, mapDispatchToProps)(Items)
      
        const fetchItems = () => { type: '@@fetchItems' }
        const fetchItemsSuccess = items =>
          { type: '@@fetchItems/success', payload: items }
        const fetchItemsError = error =>
          { type: '@@fetchItems/error', payload: error }
      
        const defaultState = { items: [], error: null }
        function reducer (state = defaultState, action) {
          switch (action.type) {
            case '@@fetchItems/success' {
              return { ...state, items: action.payload }
            }
            case '@@fetchItems/error' {
              return { ...state, error: action.error }
            }
          }
        }

Theses

  • entities in both examples are the same (state, actions, reducers, selectors, subscriptions)
  • dataflow for both platforms is the same
  • side-effect model (and code!) for both platforms is the same
  • main difference is in rendering and data binding (react jsx & virtual dom vs angular templates & shadow dom, react props vs angular input/output decorators, etc)
  • there are mostly syntactic (and implementation) differences between platforms

Theses

  • actions (or changes, or events) are stream
  • data is stream
  • purity achieved as side-effects isolated
  • complex async flow is easier to implement as Rx introduced
